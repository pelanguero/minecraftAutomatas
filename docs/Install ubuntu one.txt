#!/bin/bash

# Install docker
apt-get update
apt-get install \
    ca-certificates \
    curl \
    gnupg \
    lsb-release
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

echo \
  "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | tee /etc/apt/sources.list.d/docker.list > /dev/null

apt-get update
apt-get install docker-ce docker-ce-cli containerd.io -y
apt-get install docker-compose -y

cd /root
docker network create minecraftLegionNetwork
git clone https://github.com/sefirosweb/minecraftLegion.git
cd minecraftLegion/

echo "
const config = {
  server: 'sefirosweb.es',
  port: '25565',
  masters: [{ name: 'Sefirosweb' }],
  webServer: 'http://164.92.159.152',
  webServerPort: '4001',
  webServerPassword: 'admin',
  debugMode: false,
  customStart: false,
  autoRestart: true
}
module.exports = config
" > config.js

docker run --rm -v $PWD:/home/app -w /home/app node:16.13-bullseye npm install

docker run -d --restart always \
--name Miner9 \
--hostname Miner9 \
--network minecraftLegionNetwork \
-v $PWD:/home/app -w /home/app  \
node:16.13-bullseye node start_bot.js Miner9 legion

docker run -d --restart always \
--name Miner10 \
--hostname Miner10 \
--network minecraftLegionNetwork \
-v $PWD:/home/app -w /home/app  \
node:16.13-bullseye node start_bot.js Miner10 legion

docker run -d --restart always \
--name Miner11 \
--hostname Miner11 \
--network minecraftLegionNetwork \
-v $PWD:/home/app -w /home/app  \
node:16.13-bullseye node start_bot.js Miner11 legion

docker run -d --restart always \
--name Miner12 \
--hostname Miner12 \
--network minecraftLegionNetwork \
-v $PWD:/home/app -w /home/app  \
node:16.13-bullseye node start_bot.js Miner12 legion
